(	"context"	"errors"	"time"	goredis "github.com/redis/go-redis/v9"	"go-job-platform/internal/repository")const (	taskQueueKey  = "queue:task:pending"	retryQueueKey = time.Duration) ([]byte, error) {	res, err := r.rdb.BRPop(ctx, timeout, taskQueueKey).Result()	if err != nil {		if errors.Is(err, goredis.Nil) {			return nil, nil		}		return nil, err	}	if len(res) != 2 {		return nil, nil	}	return []byte(res[1]), nil}func (r *QueueRepository) EnqueueRetryTask(ctx context.Context, msg []byte) error {	return r.rdb.LPush(ctx, retryQueueKey, msg).Err()}```---## `internal/repository/redis/cache_repo.go`???????? worker ???????????```gopackage redisimport (	"context"	"time"	goredis (	"context"	"encoding/json"	"errors"	"log"	"time"	"go-job-platform/internal/domain"	"go-job-platform/internal/executor"	"go-job-platform/internal/queue"	"go-job-platform/internal/repository")type workerService struct {	taskRepo      repository.TaskRepository	executionRepo repository.ExecutionRepository	queueRepo     repository.QueueRepository	cacheRepo     repository.CacheRepository	executors     []executor.Executor	workerID      string	pollTimeout   time.Duration}func NewWorkerService(	taskRepo repository.TaskRepository,	executionRepo repository.ExecutionRepository,	queueRepo repository.QueueRepository,	cacheRepo repository.CacheRepository,	executors []executor.Executor,	workerID string,) WorkerService {	return &workerService{		taskRepo:      taskRepo,		executionRepo: executionRepo,		queueRepo:     queueRepo,		cacheRepo:     cacheRepo,		executors:     executors,		workerID:      workerID,		pollTimeout:   5 * time.Second,	}}func (s *workerService) ConsumeLoop(ctx context.Context) error {	for {		select {		case <-ctx.Done():			return ctx.Err()		default:		}		raw, err := s.queueRepo.DequeueTask(ctx, s.pollTimeout)		if err != nil {			log.Printf("dequeue task error: %v", err)			time.Sleep(2 * time.Second)			continue		}		if len(raw) == 0 {			continue		}		if err := s.HandleOneMessage(ctx, raw, s.workerID); err != nil {			log.Printf("handle task message error: %v", err)		}	}}func (s *workerService) HandleOneMessage(ctx context.Context, raw []byte, workerID string) error {	var msg queue.TaskMessage	if err := json.Unmarshal(raw, &msg); err != nil {		return err	}	task, err := s.taskRepo.GetByID(ctx, msg.TaskID)	if err != nil {		return err	}	if task == nil {		return errors.New("task not found")	}	if task.Status != domain.TaskStatusActive {		return errors.New("task is not active")	}	if err := s.executionRepo.UpdateStatus(ctx, msg.ExecutionNo, domain.ExecutionStatusRunning, workerID); err != nil {		return err	}	_ = s.cacheRepo.SetExecutionStatus(ctx, msg.ExecutionNo, string(domain.ExecutionStatusRunning), 30*time.Minute)	exec := s.pickExecutor(task.TaskType)	if exec == nil {		errMsg := "unsupported task type"		exitCode := -1		return s.executionRepo.Finish(			ctx,			msg.ExecutionNo,			domain.ExecutionStatusFailed,			0,			&exitCode,			&errMsg,			nil,		)	}	timeout := task.TimeoutSeconds	if timeout <= 0 {		timeout = 60	}	execCtx, cancel := context.WithTimeout(ctx, time.Duration(timeout)*time.Second)	defer cancel()	start := time.Now()	result := exec.Execute(execCtx, *task)	durationMs := time.Since(start).Milliseconds()	finalStatus := domain.ExecutionStatusSuccess	if execCtx.Err() == context.DeadlineExceeded {		finalStatus = domain.ExecutionStatusTimeout		timeoutMsg := "execution timeout"		result.ErrMsg = &timeoutMsg	} else if result.ErrMsg != nil {		finalStatus = domain.ExecutionStatusFailed	}	output := result.OutputLog	if err := s.executionRepo.Finish(		ctx,		msg.ExecutionNo,		finalStatus,		durationMs,		result.ExitCode,		result.ErrMsg,		&output,	); err != nil {		return err	}	_ = s.cacheRepo.SetExecutionStatus(ctx, msg.ExecutionNo, string(finalStatus), 24*time.Hour)	return nil}func (s *workerService) pickExecutor(taskType domain.TaskType) executor.Executor {	for _, e := range s.executors {		if e.Supports(taskType) {			return e		}	}	return nil}```---# ???worker main## `cmd/worker/main.go`????? `api-server` ????????```gopackage mainimport
